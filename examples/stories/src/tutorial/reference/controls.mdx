import { ControlTypes } from '@component-controls/core';
import { Story, Playground, PropsTable } from '@component-controls/blocks';

The [controls api](/tutorial/reference/controls-api) comes with a selection of pre-defined controls that can be used to create interactive examples of your components.

## ControlTypes.TEXT

```jsx
userName: {
  type: ControlTypes.TEXT,
  label: 'Name',
  value: 'Storyteller',
}
```

all properties of [ControlType](/tutorial/reference/controls-api/#controltype) in addition to:

| Name           | Type                                                           | Description                                                                                                                                           |
| -------------- | -------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| `escapeValue`  | boolean                                                        | allows to receive escaped string values to help prevent XSS attacks by default - false                                                                |
| `placeholder`  | string                                                         | placeholder for empty properties either undefined initial value or user clears the field                                                              |
| `rows`         | number                                                         | number of rows for a TextArea field for longer text by default, only 1 row = means a Input field \> 1 rows = an area field                            |

<Playground>
  <Story name="text-control" controls={{ text: 'some text' }}>
    {({ text }) => (
      <div>{text}</div>
    )}
  </Story>
</Playground>

## ControlTypes.NUMBER

```jsx
age: {
  type: ControlTypes.NUMBER,
  label: 'Age',
  value: 78,
  range: true,
  min: 0,
  max: 90,
  step: 5,
},
```   

all properties of [ControlType](/tutorial/reference/controls-api/#controltype) in addition to:

| Name           | Type                                                           | Description                                                                                                                                           |
| -------------- | -------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| `max`          | number                                                         | maximum allowed value for numeric property                                                                                                            |
| `min`          | number                                                         | minimum allowed value for numeric property                                                                                                            |
| `range`        | boolean                                                        | if true, will display a range type slider editor                                                                                                      |
| `step`         | number                                                         | stepper for numeric editor /i nc/dec value                                                                                                            |

<Playground>
  <Story name="numeric-control" controls={{ number: { type: ControlTypes.NUMBER, value: 7} }}>
    {({ number }) => (
      <div>{number}</div>
    )}
  </Story>
</Playground>

## ControlTypes.OPTIONS

```jsx
fruit: {
  type: ControlTypes.OPTIONS,
  label: 'Fruit',
  value: 'apple',
  options: {
    Apple: 'apple',
    Banana: 'banana',
    Cherry: 'cherry',
  },
},
```  

all properties of [ControlType](/tutorial/reference/controls-api/#controltype) in addition to:

| Name           | Type                                                                                 | Description                                                                                                                                           |
| -------------- | ------------------------------------------------------------------------------------ | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| `options*`     | Record<string, number \| string[] \| number[] \| { label: string; value: any }>      |                                                                                                                                                       |
| `display`      | 'select' \| 'multi-select' \| 'radio' \| 'inline-radio' \| 'check' \| 'inline-check' | how to render selecting the options: default is 'select'                                                                                              |

<Playground>
  <Story 
    name="options-control"
    controls={{ 
      options: { 
        type: ControlTypes.OPTIONS, 
        options: {
          Kiwi: 'kiwi',
          Guava: 'guava',
          Watermelon: 'watermelon',
        },
        display: 'check',
        value: ['watermelon'],
      } 
    }}>
    {({ options }) => (
      <div>{options.join(', ')}</div>
    )}
  </Story>
</Playground>

## ControlTypes.OBJECT

```jsx
style: {
  type: ControlTypes.OBJECT,
  label: 'Styles',
  value: {
    // do not randomize the border style
    border: { type: ControlTypes.TEXT, value: '2px dashed silver', data: null },
    borderRadius: { type: ControlTypes.NUMBER, value: 10 },
    padding: { type: ControlTypes.NUMBER, value: 10 },
  },
}
```

all properties of [ControlType](/tutorial/reference/controls-api/#controltype) in addition to:

| Name           | Type                                                           | Description                                                                                                                                           |
| -------------- | -------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| `editLabel`    | string                                                         | the label for the editor button                                                                                                                       |

<Playground>
  <Story name="object-control" controls={{ style: {
      type: ControlTypes.OBJECT,
      value: {
        color: { type: ControlTypes.COLOR, value: 'red' },
        backgroundColor: { type: ControlTypes.COLOR, value: 'lightgrey' },
        padding: { type: ControlTypes.NUMBER, value: 2, step: 5, label: 'padding (px.)' },
      },
    }}}>
    {({ style }) => (
      <button style={style}>change my style</button>
    )}
  </Story>
</Playground>

## ControlTypes.ARRAY

```jsx
arrayItems: {
  type: ControlTypes.ARRAY,
  label: 'Items',
  rowType: {
    name: { type: ControlTypes.TEXT },
  },
  value: [{ name: 'Laptop' }, { name: 'Book' }, { name: 'Whiskey' }],
},
```

all properties of [ControlType](/tutorial/reference/controls-api/#controltype) in addition to:

| Name           | Type                                                           | Description                                                                                                                                           |
| -------------- | -------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| `rowType*`     | [ComponentControls](#componentcontrols)                        | type of the items in each row of the array                                                                                                            |
| `editLabel`    | string                                                         | the label for the editor button                                                                                                                       |

<Playground>
  <Story name="array-control" controls={{ array: {
      type: ControlTypes.ARRAY,
      value: [
        { label: 'Button 1', color: 'red' },
        { label: 'Button 2', color: 'grey' },
      ],
      rowType: {
        label: { type: ControlTypes.TEXT },
        color: { type: ControlTypes.COLOR },
      },
    }}}>
    {({ array }) => (
      <div>
        {array.map(({ label, color }, index) => (
          <div key={`button_${index}`}>
            <button style={{ color }}>{label}</button>
          </div>
        ))}
      </div>    
    )}
  </Story>
</Playground>
